import re
import os

def validate_username(username):
    pattern = r'^[a-zA-Z][a-zA-Z0-9._]*@[a-zA-Z]+\.[a-zA-Z]+$'
    if re.match(pattern, username) and '@.' not in username and not username[0].isdigit() and not username[0] in "!@#$%^&*()":
        return True
    return False

def validate_password(password):
    if 6 <= len(password) <= 16:
        has_upper = any(ch.isupper() for ch in password)
        has_lower = any(ch.islower() for ch in password)
        has_digit = any(ch.isdigit() for ch in password)
        has_special = any(ch in "!@#$%^&*()-_+=" for ch in password)
        if has_upper and has_lower and has_digit and has_special:
            return True
    return False

def store_credentials(username, password, filepath='credentials.txt'):
    with open(filepath, 'a') as file:
        file.write(f"{username}:{password}\n")

def load_credentials(filepath='credentials.txt'):
    if not os.path.exists(filepath):
        return {}
    with open(filepath, 'r') as file:
        data = file.readlines()
    return {line.strip().split(':')[0]: line.strip().split(':')[1] for line in data}

def register():
    while True:

        username = input("Enter username (email): ")
        if not validate_username(username):
            print("Invalid username format.")
            continue
        password = input("Enter password: ")
        if not validate_password(password):
            print("Password does not meet criteria.")
            continue
        credentials = load_credentials()
        if username in credentials:
            print("User already registered.")
            return
        store_credentials(username, password)
        print("Registration successful!")
        return

def login():
    credentials = load_credentials()
    username = input("Enter username: ")
    password = input("Enter password: ")
    if username in credentials and credentials[username] == password:
        print("Login successful. Welcome!")
    else:
        print("Invalid credentials.")
        choice = input("Do you want to (R)egister or (F)orget Password? ").lower()
        if choice == 'r':
            register()
        elif choice == 'f':
            forgot_password(credentials, username)

def forgot_password(credentials, username):
    if username in credentials:
        print(f"Your password is: {credentials[username]}")
        choice = input("Do you want to reset your password? (y/n): ").lower()
        if choice == 'y':
            while True:
                new_password = input("Enter new password: ")
                if validate_password(new_password):
                    credentials[username] = new_password
                    with open('credentials.txt', 'w') as file:
                        for user, pwd in credentials.items():
                            file.write(f"{user}:{pwd}\n")
                    print("Password updated successfully.")
                    break
                else:
                    print("New password does not meet criteria.")
    else:
        print("Username not found. Please register.")
        register()

def main():
    print("Welcome to the Python Login System!")
    while True:
        action = input("Choose action - Register (R), Login (L), Quit (Q): ").lower()
        if action == 'r':
            register()
        elif action == 'l':
            login()
        elif action == 'q':
            print("Exiting program.")
            break
        else:
            print("Invalid option. Please try again.")

main()
